#include <stdio.h>

main () {
  int fahr, celcius;
  int lower, upper, step;

  lower = 0;
  upper = 300;
  step = 20;

  fahr = lower;

  while (fahr <= upper) {
    celcius = 5 * (fahr - 32) / 9;
    printf("%d : %d\n", fahr, celcius);
    fahr = fahr + step;
  }
}

if / else condition
-------------------


#include <stdio.h>

int power (int, int);

int main () {
  printf("%d\n", power(3, 4));
  return 0;
}

int power (int base, int n) {
  int i, p;
  p = 1;

  if (base < 0) {
    return 0;
  } else {
    for (i = 1; i <= n; ++i)
      p = p * base

    return p;
  }
}

void fizzbuzz(int limit)
{
    int i;

    for (i=1; i<=limit; ++i) {
        if (!(i % 15))
            printf("FizzBuzz");
        else if (!(i % 3))
            printf("Fizz");
        else if (!(i % 5))
            printf("Buzz");
        else
            printf("%d", i);

        if (i < limit)
            printf(" ");
    }
}

switch case
-----------


#include <stdio.h>

int main () {
  printf("%d", chars('a'));
  return 0;
}

int chars (char c) {
  int p;

  switch (c) {
    case 'a':
      p = 0;
      break;
    case 'b':
      p = 1;
      break;
    default:
      p = 3;
      break;
  }
  return p;
}

void number_printer(int number)
{

  switch(number)
  {
    case 0:
      printf ("Zero");
      break;
    case 1:
      printf ("One");
      break;
    case 2:
      printf ("Two");
      break;
    case 3:
      printf ("Three");
      break;
    default:
      printf("%d", number);
  }
}
